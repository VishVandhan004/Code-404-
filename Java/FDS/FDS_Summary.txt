/*Problem Statement-1:
Arun wants to design a Right Angle Star Pattern.
Let’s help him by accepting a number.

Example 1
Input:
3
OUTPUT:
*
* *
* * *

*/
import java.util.*;
class RightTrianglePattern
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);
        int size = sc.nextInt();
        for(int i=0;i<size;i++)
        {
            for(int j=0;j<=i;j++)
            {
                System.out.print("* ");
            }
                System.out.println();
        }
    }
}
/*Problem Statement-2:
Varun wants to design a LeftTrianglePattern.
Let’s help him by accepting a number.

Input:
5
Output: 
           *
         * *
       * * *
     * * * *
   * * * * *
   
*/
import java.util.*;
class LeftTrianglePattern{
    public static void main(String args[])
    {
         Scanner sc = new Scanner(System.in);
        int size = sc.nextInt();
        for(int i=0;i<size;i++)
        {
            for(int j=5;j>i;j--)
            {
                System.out.print(" ");
            }
            for(int j=0;j<i;j++)
            {
                System.out.print("*");
            }
            System.out.println();
        }
    }
}
