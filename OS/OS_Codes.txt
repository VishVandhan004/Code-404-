/* write a programme to print the reverse of a number
sample input 
Enter an integer: 123                                                                                                   
Reversed number = 321   */      

#include <stdio.h>
int main() {
  int n;
  printf("Enter an integer: ");
  scanf("%d",&n);
  int rev=0;
  while(n>0)
  {
      int temp=n%10;
      rev=(rev*10)+temp;
      n=n/10;
  }
  printf("Reversed number = %d",rev);
}

/* wrire a program to find factorial of a given number
input=
Enter an integer: 5    
output=
Factorial of 5 = 120 */

#include<stdio.h>
int fact(int n)
{
    if(n==1 || n==0)
    {
        return 1;
    }
    else
    {
        return n*fact(n-1);
    }
}
int main()
{
    int n;
    printf("Enter an integer: ");
    scanf("%d",&n);
    printf("Factorial of %d = %d",n,fact(n));
}
